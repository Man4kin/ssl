Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	Свойства = ДатыЗапретаИзмененияПовтИсп.СвойстваРазделов();
	
	// Установка варианта отчета
	Если Параметры.ПоРазделамОбъектам = Истина Тогда
		
		Если Свойства.ПоказыватьРазделы И НЕ Свойства.ВсеРазделыБезОбъектов Тогда
			УстановитьТекущийВариант("ПоРазделамОбъектам");
			
		ИначеЕсли Свойства.ВсеРазделыБезОбъектов Тогда
			УстановитьТекущийВариант("ПоРазделам");
		Иначе
			УстановитьТекущийВариант("ПоОбъектам");
		КонецЕсли;
		
	ИначеЕсли Параметры.ПоРазделамОбъектам = Ложь Тогда
		
		Если Свойства.ПоказыватьРазделы И НЕ Свойства.ВсеРазделыБезОбъектов Тогда
			УстановитьТекущийВариант("ПоПользователям");
			
		ИначеЕсли Свойства.ВсеРазделыБезОбъектов Тогда
			УстановитьТекущийВариант("ПоПользователямБезОбъектов");
		Иначе
			УстановитьТекущийВариант("ПоПользователямБезРазделов");
		КонецЕсли;
	Иначе
		УстановитьТекущийВариант("ПоПользователям");
	КонецЕсли;
