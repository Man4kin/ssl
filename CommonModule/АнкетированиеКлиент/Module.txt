// Создает структуру отбора, для дальнейшей передачи на сервер,
//и использования в качестве параметров отбора динамических списков вызываемых форм
//
Функция СоздатьСтруктуруПараметраОтбора(ТипОтбора,ЛевоеЗначение,ВидСравнения,ПравоеЗначение) Экспорт

	 СтруктураВозврата = Новый Структура;
	 СтруктураВозврата.Вставить("ТипОтбора",ТипОтбора);
	 СтруктураВозврата.Вставить("ЛевоеЗначение",ЛевоеЗначение);
	 СтруктураВозврата.Вставить("ВидСравнения",ВидСравнения);
	 СтруктураВозврата.Вставить("ПравоеЗначение",ПравоеЗначение);
	 
	 Возврат СтруктураВозврата;

КонецФункции // ()

// Формирует структуру параметров для передачи в вспомогательную форму ввода текста
//
Функция СформироватьСтруктуруПараметровДляФормыВводаТекста(Форма,Элемент) Экспорт

	СтруктураПараметров = Новый Структура;
	Если ТипЗнч(Элемент.Родитель) = Тип("ГруппаФормы") Тогда
	    СтруктураПараметров.Вставить("Текст",Форма[Элемент.Имя]);
		СтруктураПараметров.Вставить("ИмяТаблицы",Элемент.Имя);
	Иначе	
		СтруктураПараметров.Вставить("Текст",Форма[Элемент.Родитель.Имя][Элемент.Родитель.ТекущаяСтрока][Элемент.Имя]);
		СтруктураПараметров.Вставить("ИмяТаблицы",Элемент.Родитель.Имя);
		СтруктураПараметров.Вставить("НомерСтроки",Элемент.Родитель.ТекущаяСтрока);
		СтруктураПараметров.Вставить("ИмяКолонки",Элемент.Имя);
	КонецЕсли;
	
	Возврат СтруктураПараметров;

КонецФункции // ()
